### 1.3 CSMA/CA Protocol

```python
# ============================================
# CSMA/CA Implementation
# ============================================

import random
import time

class WiFiNode:
    def __init__(self, node_id):
        self.node_id = node_id
        self.channel_busy = False
        self.cw_min = 15  # Contention Window
        self.cw_max = 1023
        self.cw = self.cw_min
    
    def sense_channel(self):
        """حساسية القناة"""
        # في الواقع: قياس الطاقة المستلمة
        return self.channel_busy
    
    def backoff(self):
        """التراجع عشوائي"""
        # اختيار وقت عشوائي من CW
        backoff_time = random.randint(0, self.cw)
        
        # انتظار
        time.sleep(backoff_time)
        
        # التحقق مرة أخرى
        if self.sense_channel():
            # القناة مشغولة - زيادة CW وإعادة المحاولة
            self.cw = min(2 * self.cw + 1, self.cw_max)
            return False
        else:
            # القناة متاحة - الإرسال
            return True
    
    def transmit(self, data):
        """الإرسال مع CSMA/CA"""
        # Sense initially
        if self.sense_channel():
            # Wait for DIFS
            time.sleep(50e-6)  # DIFS = 50 μs
            
            # حادثة المحاولة
            while self.sense_channel():
                time.sleep(50e-6)
        
        # Backoff
        if not self.backoff():
            # المحاولة فشلت - will retry later
            return False
        
        # Transmit
        self._send_frame(data)
        
        # Wait for ACK
        if self._wait_for_ack():
            # نجاح
            self.cw = self.cw_min  # Reset CW
            return True
        else:
            # فشل - زيادة CW
            self.cw = min(2 * self.cw, self.cw_max)
            return False
    
    def _send_frame(self, data):
        """إطار الإرسال"""
        pass
    
    def _wait_for_ack(self):
        """ال انتظار لـ ACK"""
        time.sleep(300e-6)  # ACK timeout = 300 μs
        return True
```

### 2.5 Massive MIMO

$$y = \sqrt{\frac{P}{M}} \mathbf{H} \mathbf{w} s + \mathbf{n}$$

```python
# ============================================
# Massive MIMO System
# ============================================

import numpy as np

class MassiveMIMO:
    def __init__(self, n_antennas, n_users):
        self.n_antennas = n_antennas  # M
        self.n_users = n_users    # K
        self.H = None  # Channel matrix
    
    def generate_channel(self):
        """توليد نموذج القناة"""
        # Rayleigh fading
        self.H = np.sqrt(0.5) * (
            np.random.randn(self.n_antennas, self.n_users) + 
            1j * np.random.randn(self.n_antennas, self.n_users)
        )
        return self.H
    
    def zero_forcing(self, s):
        """ZF Beamforming"""
        # Compute ZF matrix
        H = self.H
        W = np.linalg.inv(H.T @ H) @ H.T
        
        # Apply beamforming
        x = W @ s
        
        # Normalize power
        x = x / np.linalg.norm(x) * np.sqrt(self.n_antennas)
        
        return x
    
    def detect(self, y):
        """اكتشاف الإشارة"""
        # ZF detection
        H_pinv = np.linalg.pinv(self.H)
        s_hat = H_pinv @ y
        
        return s_hat
```

```python
# ============================================
# AODV (Ad-hoc On-Demand Distance Vector)
# ============================================

class AODVNode:
    def __init__(self, node_id):
        self.node_id = node_id
        self.routing_table = {}  # destination -> (next_hop, hop_count, seq_num)
        self.neighbors = set()
    
    def discover_route(self, destination):
        """اكتشاف المسار عند الطلب"""
        # RREQ (Route Request) broadcast
        rreq = {
            'type': 'RREQ',
            'source': self.node_id,
            'destination': destination,
            'broadcast_id': self._get_broadcast_id(),
            'hop_count': 0
        }
        
        self._broadcast(rreq)
        # Will wait for RREP
        return self._wait_for_route(destination)
    
    def process_rreq(self, rreq):
        """معالجة RREQ"""
        # Update reverse path
        source = rreq['source']
        self.routing_table[source] = (
            rreq['from'],  # next hop
            rreq['hop_count'],
            rreq.get('source_seq', 0)
        )
        
        # Check if we know route to destination
        if rreq['destination'] in self.routing_table:
            # Send RREP (Route Reply)
            rrep = {
                'type': 'RREP',
                'source': self.node_id,
                'destination': rreq['source'],
                'destination_seq': self.routing_table[rreq['destination']][2]
            }
            self._send_to(rreq['from'], rrep)
        else:
            # Forward RREQ
            rreq['hop_count'] += 1
            self._broadcast(rreq)
```

### 3.3 DSR (Dynamic Source Routing)

```python
# ============================================
# DSR Protocol
# ============================================

class DSRNode:
    def __init__(self, node_id):
        self.node_id = node_id
        self.route_cache = {}  # destination -> [path]
        self.send_buffer = []
    
    def route_discovery(self, destination):
        """اكتشاف المسار"""
        route_request = {
            'id': self._get_request_id(),
            'source': self.node_id,
            'target': destination,
            'path': [self.node_id]
        }
        
        self._broadcast(route_request)
        return self._wait_for_route(destination)
    
    def process_route_request(self, rreq):
        """معالجة طلب المسار"""
        if rreq['target'] == self.node_id:
            # Found destination, send reply
            route_reply = {
                'type': 'RREP',
                'route': rreq['path'][::-1]  # Reverse path
            }
            next_hop = rreq['path'][-2] if len(rreq['path']) > 1 else None
            if next_hop:
                self._send_to(next_hop, route_reply)
        elif rreq['target'] not in self.route_cache:
            # Forward request
            rreq['path'].append(self.node_id)
            self._broadcast(rreq)
```

# LEACH Protocol
# ============================================

import random
import math

class LEACHNode:
    def __init__(self, node_id, x, y, energy=1.0):
        self.node_id = node_id
        self.x = x
        self.y = y
        self.energy = energy
        self.is_cluster_head = False
        self.cluster_id = None
    
    def become_cluster_head(self, threshold):
        """قرر ما إذا كنت Cluster Head"""
        # Probability based threshold
        p = threshold * (self.energy / 1.0)  # Adjust for energy
        
        if random.random() < p:
            self.is_cluster_head = True
            return True
        return False
    
    def receive_from_cluster(self, data):
        """البيانات من Cluster"""
        self.energy -= self._calculate_energy(data)
    
    def compute_distance(self, other):
        """المسافة"""
        return math.sqrt((self.x - other.x)**2 + (self.y - other.y)**2)
    
    def _calculate_energy(self, data_size):
        """حساب طاقة الإرسال"""
        # E_tx = E_elec * d + E_amp * d^2
        d = data_size
        return 50e-9 * d + 100e-12 * d**2


class LEACHProtocol:
    def __init__(self, nodes, n_clusters):

        self.n_clusters = n_clusters
        self.clusters = {}

## 5. تقنيات modulation وتقنيات الإرسال

### 5.1 OFDM (Orthogonal FDM)

$$s(t) = \sum_{k=0}^{N-1} c_k e^{j2\pi f_k t}$$

```python
# ============================================
# OFDM Implementation
# ============================================

import numpy as np

class OFDMSystem:
    def __init__(self, n_subcarriers):
        self.n_subcarriers = n_subcarriers
    
    def modulate(self, data_bits):
        """OFDM Modulation"""
        # Map to constellation (QPSK example)
        symbols = self._map_to_constellation(data_bits)
        
        # IFFT
        ofdm_symbol = np.fft.ifft(symbols)
        
        # Add CP (Cyclic Prefix)
        n_cp = self.n_subcarriers // 4
        ofdm_symbol_with_cp = np.concatenate([
            ofdm_symbol[-n_cp:],  # Copy of last CP samples
            ofdm_symbol
        ])
        
        return ofdm_symbol_with_cp
    
    def demodulate(self, received_signal):
        """OFDM Demodulation"""
        # Remove CP
        n_cp = self.n_subcarriers // 4
        ofdm_symbol = received_signal[n_cp:]
        
        # FFT
        symbols = np.fft.fft(ofdm_symbol)
        
        # Detect
        bits = self._demap_constellation(symbols)
        
        return bits
    
    def _map_to_constellation(self, bits):
        """QPSK Mapping"""
        # Simplified: 2 bits per symbol
        mapping = {
            '00': 1+1j, '01': -1+1j,
            '11': -1-1j, '10': 1-1j
        }
        return [mapping[bits[i:i+2]] for i in range(0, len(bits), 2)]
```

### 5.2 CDMA

$$s(t) = \sum_{k=0}^{K-1} b_k(t) \cdot c_k(t)$$

```python
# ============================================
# CDMA Spreading
# ============================================

class CDMASystem:
    def __init__(self, code_length):
        self.code_length = code_length
        self.codes = self._generate_codes()
    
    def _generate_codes(self):
        """توليد spread codes (Walsh codes)"""
        # Hadamard matrix
        H = np.array([[1]])
        for _ in range(int(np.log2(self.code_length))):
            H = np.block([
                [H, H],
                [H, -H]
            ])
        return H
    
    def spread(self, data, user_id):
        """spread data"""
        code = self.codes[user_id]
        spread_signal = data * code
        return spread_signal
    
    def despread(self, received_signal, user_id):
        """استرجاع البيانات"""
        code = self.codes[user_id]
        # Correlate
        despread = np.sum(received_signal * code) / self.code_length
        return np.sign(despread.real)
```

```python
# ============================================
# WPA3 Handshake
# ============================================

class WPA3Handshake:
    def __init__(self):
        self.password = None
    
    def authenticate(self, username, password):
        """SAE Authentication"""
        # Password authenticated key exchange
        self.password = password
        
        # Send commit
        commit = self._generate_commit(password)
        
        # Process peer commit
        peer_commit = self._receive()
        
        # Derive keys
        keys = self._derive_keys(commit, peer_commit)
        
        return keys
    
    def _generate_commit(self, password):
        """توليد commit"""
        # Scalar multiplication in finite field
        return password
    
    def _derive_keys(self, commit, peer_commit):
        """استخراج المفاتيح"""
        # PMK (Pairwise Master Key)
        pmk = hashlib.pbkdf2(
            password,
            commit + peer_commit,
            iterations=4096,
            keylen=32
        )
        return pmk
```
